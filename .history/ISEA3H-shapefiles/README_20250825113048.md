This repository contains shapefiles for the ISEA3H grid at various resolutions. The shapefiles were generated using R and the `sf` package. 

These shapefiles are intended to be used with GBIF SQL downloads. 




```R
library(rgbif)
library(sf)

# Plot some of the grids for README
library(sf)
library(dplyr)
library(ggplot2)


sql <-
"
SELECT 
  GBIF_ISEA3HCode(
    6, 
    decimalLatitude,
    decimalLongitude,
    COALESCE(coordinateUncertaintyInMeters, 1000) 
  ) AS geogrid_id,
  COUNT(DISTINCT speciesKey) AS unique_species_count
FROM
  occurrence
GROUP BY
  mgrs
"

sf_obj <- resolution_1 <- st_read("ISEA3H-shapefiles/resolution-6/ISEA3H-6.shp")
sf_obj |> glimpse()

p <- ggplot() +
geom_sf(data=sf_obj)

ggsave(p, filename = "ISEA3H-shapefiles/grid_plot.svg")




```
